\subsection{电容桥}
$f=1.000\textrm{kHz}$，$E=4.01\textrm V$，$\frac{e_f}{f}=0.1\%$
\subsubsection{纸质电容（$0.2\mu \textrm F$）}
$R_1=R_2=800.0\Omega$，$R_0=2.8\Omega$，$C_0=0.2188\mu \textrm F$，最终$\Delta U=0.13\textrm{mV}$。
$$\frac{e_{R_1}}{R_1}=\frac{e_{R_2}}{R_2}=0.1\%$$
$$\frac{e_{R_0}}{R_0}=\frac{2\times0.5\%+0.8\times2\%}{2.8}=0.9\%$$
$$\frac{e_{C_0}}{C_0}=\frac{0.2\times0.5\%+0.01\times0.65\%+0.008\times2\%+0.0008\times5\%}{0.2188}=0.6\%$$

计算电容$C_x$：
$$C_x=\frac{R_2}{R_1}C_0=0.2188\mu \textrm F$$
$$\sigma_{C_x}=\frac{C_x}{\sqrt3}\sqrt{(\frac{e_{R_1}}{R_1})^2+(\frac{e_{R_2}}{R_2})^2+(\frac{e_{C_0}}{C_0})^2}=\frac{0.22}{\sqrt3}\times0.6\%=0.0008\mu \textrm F$$
$$C_x\pm\sigma_{C_x}=(0.2188\pm0.0008)\mu \textrm F$$

计算损耗电阻$R_C$：
$$R_C=\frac{R_1}{R_2}R_0=2.8\Omega$$
$$\sigma_{R_C}=\frac{R_C}{\sqrt3}\sqrt{(\frac{e_{R_1}}{R_1})^2+(\frac{e_{R_2}}{R_2})^2+(\frac{e_{R_0}}{R_0})^2}=\frac{2.8}{\sqrt3}\times0.9\%=0.02\Omega$$
$$R_C\pm\sigma_{R_C}=(2.80\pm0.02)\Omega$$

计算损耗角$\tan{\delta}$：
$$\tan{\delta}=2 \pi R_0 C_0 f=3.85\times10^{-3}$$
$$\sigma_{\tan{\delta}}=\frac{\tan{\delta}}{\sqrt3}\sqrt{(\frac{e_{R_0}}{R_0})^2+(\frac{e_{C_0}}{C_0})^2+(\frac{e_f}{f})^2}=\frac{3.9\times10^{-3}}{\sqrt3}\times1\%=0.02\times10^{-3}$$
$$\tan{\delta}\pm\sigma_{\tan{\delta}}=(3.85\pm0.02)\times10^{-3}$$

\subsection{麦克斯韦桥测空心电感（$L=6\textrm{mH}$，$R_L=20\Omega$）}
$R_1=500.0\Omega$，$R_0=16.3\Omega$，$R_2=500.0\Omega$，$L_0=6.0\textrm{mH}$，$R_{L_0}=4.48\Omega$，最终$\Delta U=0.41\textrm{mV}$。
$$\frac{e_{R_1}}{R_1}=\frac{e_{R_2}}{R_2}=0.1\%$$
$$\frac{e_{R_0}}{R_0}=\frac{10\times0.1\%+6\times0.5\%+0.3\times2\%}{16.3}=0.3\%$$
$$\frac{e_{L_0}}{L_0}=2\%$$
$$\frac{e_{R_{L_0}}}{R_{L_0}}=\frac{0.01}{4.48}=0.2\%$$

计算电感$L_x$：
$$L_x=\frac{R_1 }{R_2}L_0=6.00\textrm{mH}$$
$$\sigma_{L_x}=\frac{L_x}{\sqrt3}\sqrt{(\frac{e_{R_1}}{R_1})^2+(\frac{e_{R_2}}{R_2})^2+(\frac{e_{L_0}}{L_0})^2}=\frac{6}{\sqrt3}\times2\%=0.07\textrm{mH}$$
$$L_x\pm\sigma_{L_x}=(6.00\pm0.07)\textrm{mH}$$

计算损耗电阻$R_L$：
$$R_L=\frac{R_1 }{R_2}(R_0+R_{L_0})=20.78\Omega$$
$$\sigma_{R_L}=\frac{R_L}{\sqrt3}\sqrt{(\frac{e_{R_1}}{R_1})^2+(\frac{e_{R_2}}{R_2})^2+\frac{e_{R_0}^2+e_{R_{L_0}}^2}{(R_0+R_{L_0})^2}}=\frac{21}{\sqrt3}\times0.3\%=0.04\Omega$$
$$R_L\pm\sigma_{R_L}=(20.78\pm0.04)\Omega$$

计算$Q$值：
$$Q=2 \pi  f \frac{L_0}{R_0+R_{L_0}}=1.81$$
$$\sigma_Q=\frac{Q}{\sqrt3}\sqrt{\frac{e_{R_0}^2+e_{R_{L_0}}^2}{(R_0+R_{L_0})^2}+(\frac{e_{L_0}}{L_0})^2+(\frac{e_f}{f})^2}=\frac{1.8}{\sqrt3}\times2\%=0.02$$
$$Q\pm\sigma_Q=1.81\pm0.02$$

\subsection{比较麦克斯韦―维恩桥和麦克斯韦桥的收敛性（含思考题（2））}
麦克斯韦―维恩桥四个臂的阻抗分别为：$\tilde{Z_1}=R_1$，$\tilde{Z_2}=\frac{R_0}{1+j\omega C_0 R_0}$，$\tilde{Z_3}=R_L+j\omega L_x$，$\tilde{Z_4}=R_2$。

电桥平衡时，有
$$\frac{\tilde{Z_1}}{\tilde{Z_2}}=\frac{\tilde{Z_3}}{\tilde{Z_4}}$$
$$(R_L+j\omega L_x)-(\frac{R_1 R_2}{R_0}+j\omega C_0 R_1 R_2)=0$$
记复数$\textbf A=R_L+j\omega L_x$，$\textbf B=\frac{R_1 R_2}{R_0}+j\omega C_0 R_1 R_2$，则$\textbf A$是一个常数。调节$R_0$可改变$\textbf B$的实部，调节$C_0$可改变$\textbf B$的虚部。
\begin{figure}[H]
\centering
\includegraphics[scale=0.1]{lab-27-figure-1.jpg}
\caption{麦克斯韦―维恩桥和麦克斯韦桥达到平衡的过程图}
\end{figure}
所以理论上，麦克斯韦―维恩桥由于$C_0$和$R_0$是独立的，只需要两步即可调平衡（如左图）；而麦克斯韦桥一般情况下需要很多步逐步趋近平衡点（如右图）。

本次实验中，由于我的待测电感值恰好为$6.00\textrm{mH}$，与电感箱$L_0$的值正好相等，也即初始点$\textbf A$与$\textbf B$的虚部相等，所以我只调了一次$R_0$（$\textbf B$的实部）就直接使电桥趋于平衡。但理论上讲，麦克斯韦桥的收敛性是较差的。

\subsection{麦克斯韦―维恩桥测量标准互感器的互感值}
顺接时，$L$顺$=L_1+L_2+2M$；反接时，$L$反$=L_1+L_2-2M$。所以$M=\frac 1 4 (L$顺$-L$反$)$。
\subsubsection{顺接}
$R_1=R_2=640.0\Omega$，$R_0=3175.5\Omega$，$C_0=0.6155\mu\textrm F$，最终$\Delta U=0.16\textrm{mV}$。
$$\frac{e_{R_1}}{R_1}=\frac{e_{R_2}}{R_2}=\frac{e_{R_0}}{R_0}=0.1\%$$
$$\frac{e_{C_0}}{C_0}=\frac{0.6\times0.5\%+0.01\times0.65\%+0.005\times2\%+0.0005\times5\%}{0.62}=0.5\%$$

计算电感$L_x1$（顺接）：
$$L_x1=C_0 R_1 R_2=0.252\textrm{H}$$
$$\sigma_{L_x1}=\frac{L_x1}{\sqrt3}\sqrt{(\frac{e_{R_1}}{R_1})^2+(\frac{e_{R_2}}{R_2})^2+(\frac{e_{C_0}}{C_0})^2}=\frac{0.62}{\sqrt3}\times0.5\%=0.002\textrm{H}$$
$$L_x1\pm\sigma_{L_x1}=(0.252\pm0.002)\textrm{H}$$

\subsubsection{反接}
$R_1=R_2=300.0\Omega$，$R_0=786.8\Omega$，$C_0=0.5895\mu\textrm F$，最终$\Delta U=0.02\textrm{mV}$。
$$\frac{e_{R_1}}{R_1}=\frac{e_{R_2}}{R_2}=\frac{e_{R_0}}{R_0}=0.1\%$$
$$\frac{e_{C_0}}{C_0}=\frac{0.5\times0.5\%+0.08\times0.65\%+0.009\times2\%+0.0005\times5\%}{0.59}=0.5\%$$

计算电感$L_x2$（反接）：
$$L_x2=C_0 R_1 R_2=0.0531\textrm{H}$$
$$\sigma_{L_x2}=\frac{L_x2}{\sqrt3}\sqrt{(\frac{e_{R_1}}{R_1})^2+(\frac{e_{R_2}}{R_2})^2+(\frac{e_{C_0}}{C_0})^2}=\frac{0.053}{\sqrt3}\times0.5\%=0.0002\textrm{H}$$
$$L_x2\pm\sigma_{L_x2}=(0.0531\pm0.0002)\textrm{H}$$

\subsubsection{互感值$M$及其不确定度$\sigma_M$}
$$M=\frac1 4 (L_x1-L_x2)=0.0497\textrm{H}$$
$$\sigma_M=\frac1 4 \sqrt{\sigma_{L_x1}^2+\sigma_{L_x2}^2}=\frac1 4 \times 0.002=0.0005\textrm{H}$$
$$M\pm\sigma_M=(0.0497\pm0.0005)\textrm{H}$$

当频率升高以后，磁畴极性的变化跟不上电流的变化，这样它的磁性就弱了，磁导率下降，因而电感量下降。

此外，频率升高时，铁芯的损耗（磁滞损耗和涡流损耗）增大，以及线圈的趋肤效应加大，导致电感器的损耗增大。
